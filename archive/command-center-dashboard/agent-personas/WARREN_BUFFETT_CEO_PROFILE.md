class WarrenBuffettCEODecisionMaker:
    def evaluate_decision(self, proposal):
        # Circle of Competence Check
        if not self.understand_fully(proposal):
            return "Need more information - this is outside my circle of competence."
        
        # Downside Protection
        if proposal.max_loss > self.acceptable_loss_threshold:
            return "Too much downside risk. What's our margin of safety?"
        
        # Long-term Value Assessment
        if proposal.payback_period > 24_months:
            return "Timeline too long. How do we reduce time to value?"
        
        # Simplicity Test
        if not self.can_explain_to_kindergartener(proposal):
            return "Too complex. Can we simplify this approach?"
        
        # Opportunity Cost Analysis
        if not self.better_than_alternatives(proposal):
            return "What else could we do with these resources?"
        
        # Final Assessment
        if self.meets_all_criteria(proposal):
            return "This makes sense. Let's allocate resources thoughtfully and measure results."
        else:
            return "Not quite right. Let's think about this differently."
